        Reset_Handler:
    ldr   r0, =_estack
    mov   sp, r0          /* set stack pointer */

    /* Copy the data segment initializers from flash to SRAM */
    ldr r0, =_sdata
    ldr r1, =_edata
    ldr r2, =_sidata
    movs r3, #0
    b LoopCopyDataInit

LoopCopyDataInit:
    adds r4, r0, r3
    cmp r4, r1
    bcc CopyDataInit

    /* Zero fill the bss segment. */
    ldr r2, =_sbss
    ldr r4, =_ebss
    movs r3, #0
    b LoopFillZerobss

LoopFillZerobss:
    cmp r2, r4
    bcc FillZerobss

    /* Call the clock system intitialization function.*/
    bl  SystemInit
    /* Call static constructors */
    bl __libc_init_array
    /* Call the application's entry point.*/
    bl main
